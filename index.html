/** * 禮券管理系統 - 極致索引版 V3.5.2 (GAS) * * 優化重點： * 1. 索引驅動 (Index-Driven)：建立「代碼-行號」地圖，實現 O(1) 級別的精準定位。 * 2. 區域性更新 (Precision Update)：僅針對受影響的儲存格進行寫入，不再重寫整張表。 * 3. 請求合併：維持單一請求處理所有邏輯。 */ const SS = SpreadsheetApp.getActiveSpreadsheet(); function doGet(e) { try { const action = e.parameter.action; if (action === 'verify') { const email = e.parameter.email; const code = e.parameter.code; const userData = SS.getSheetByName("Users").getDataRange().getValues(); for (let i = 1; i < userData.length; i++) { if (userData[i][0] === email && userData[i][1].toString() === code) { return jsonResponse({ success: true, name: userData[i][2] }); } } return jsonResponse({ success: false }); } if (action === 'getConfig') { const data = SS.getSheetByName("Config").getDataRange().getValues(); return jsonResponse(data.slice(1).map(r => ({ supplier: r[0], name: r[1], rule1: r[2], rule2: r[3], calcType: r[4] }))); } if (action === 'getBatch') { const val = SS.getSheetByName("System").getRange("A2").getValue(); return jsonResponse({ batch: val }); } if (action === 'getRecords') { const data = SS.getSheetByName("Records").getDataRange().getValues(); return jsonResponse(data.slice(1)); } if (action === 'detectInfoByCode') { const code = e.parameter.code ? e.parameter.code.toString().trim() : ""; if (!code) return jsonResponse({ success: false }); const sheet = SS.getSheetByName("GiftStatus"); const codes = sheet.getRange(1, 4, sheet.getLastRow(), 1).getValues(); for (let i = 1; i < codes.length; i++) { if (codes[i][0].toString().trim() === code) { const rowData = sheet.getRange(i + 1, 2, 1, 2).getValues()[0]; return jsonResponse({ success: true, supplier: rowData[0], giftName: rowData[1] }); } } return jsonResponse({ success: false }); } if (action === 'getStats') { const startDate = e.parameter.start ? new Date(e.parameter.start) : null; const endDate = e.parameter.end ? new Date(e.parameter.end) : null; if (endDate) endDate.setHours(23, 59, 59, 999); const fSupplier = e.parameter.supplier; const fGift = e.parameter.gift; const statusData = SS.getSheetByName("GiftStatus").getDataRange().getValues(); const statsMap = {}; for (let i = 1; i < statusData.length; i++) { const row = statusData[i]; const time = row[8] ? new Date(row[8]) : null; const supplier = row[1]; const giftName = row[2]; const actStatus = row[4]; const usedStatus = row[5]; const orderNo = row[6]; if (startDate && (!time || time < startDate)) continue; if (endDate && (!time || time > endDate)) continue; if (fSupplier && fSupplier !== "" && supplier !== fSupplier) continue; if (fGift && fGift !== "" && giftName !== fGift) continue; const key = supplier + "_" + giftName; if (!statsMap[key]) { statsMap[key] = { giftName: giftName, supplier: supplier, total: 0, used: 0, unactivated: 0, activated: 0 }; } statsMap[key].total++; if (usedStatus === "已使用" || (orderNo && orderNo !== "")) statsMap[key].used++; if (actStatus === "未啟動") statsMap[key].unactivated++; if (actStatus === "已啟動") statsMap[key].activated++; } const statsArray = Object.values(statsMap).sort((a, b) => { if (a.supplier < b.supplier) return -1; if (a.supplier > b.supplier) return 1; return a.giftName.localeCompare(b.giftName); }); return jsonResponse(statsArray); } if (action === 'checkActivation') { const start = e.parameter.start; const end = e.parameter.end; const sheet = SS.getSheetByName("GiftStatus"); const data = sheet.getDataRange().getValues(); let unactivatedCodes = []; let totalInRange = 0; for (let i = 1; i < data.length; i++) { const code = data[i][3].toString(); if (code >= start && code <= end) { totalInRange++; if (data[i][4] === "未啟動") unactivatedCodes.push(code); } } return jsonResponse({ total: totalInRange, unactivatedCount: unactivatedCodes.length, unactivatedCodes: unactivatedCodes }); } if (action === 'checkUsage') { const codes = JSON.parse(e.parameter.codes).map(c => c.toString().trim()); const sheet = SS.getSheetByName("GiftStatus"); const dbCodes = sheet.getRange(1, 4, sheet.getLastRow(), 3).getValues(); const usedCodes = []; const codeSet = new Set(codes); for (let i = 1; i < dbCodes.length; i++) { const dbCode = dbCodes[i][0].toString().trim(); if (codeSet.has(dbCode) && dbCodes[i][2] === "已使用") { usedCodes.push(dbCode); } } return jsonResponse({ usedCodes: usedCodes }); } return jsonResponse({ result: "error", message: "Unknown action" }); } catch (err) { return jsonResponse({ result: "error", message: err.toString() }); } } function doPost(e) { try { const data = JSON.parse(e.postData.contents); const mode = data.mode || data.action; if (mode === 'checkRegDuplicates') { const codes = data.codes.map(c => c.toString().trim()); const sheet = SS.getSheetByName("GiftStatus"); const dbCodes = sheet.getRange(1, 4, sheet.getLastRow(), 1).getValues(); const dbCodeSet = new Set(dbCodes.map(r => r[0].toString().trim())); const existingCodes = codes.filter(code => dbCodeSet.has(code)); return jsonResponse({ duplicates: existingCodes }); } if (mode === 'register') { const statusSheet = SS.getSheetByName("GiftStatus"); const recordSheet = SS.getSheetByName("Records"); const payload = data.payload; if (!payload || payload.length === 0) throw new Error("無效的資料負載"); const first = payload[0]; const subSheetName = first.Supplier + "_" + first.GiftName; const subSheet = getOrCreateSubSheet(subSheetName); const rows = payload.map(item => [item.Batch, item.Supplier, item.GiftName, item.Code.toString().trim(), "未啟動", "未使用", "", "", ""]); if (rows.length > 0) { statusSheet.getRange(statusSheet.getLastRow() + 1, 1, rows.length, 9).setValues(rows); subSheet.getRange(subSheet.getLastRow() + 1, 1, rows.length, 9).setValues(rows); } const last = payload[payload.length - 1]; recordSheet.appendRow([new Date(), first.Supplier, first.GiftName, first.Code, last.Code, payload.length, data.user || "系統", "入庫登記", "N/A", "正常", "登記"]); return jsonResponse({ result: "success", count: payload.length }); } // 極致優化出庫模式：索引驅動 + 區域寫入 if (mode === 'use') { const payload = data.payload; const user = data.user || ""; const now = new Date(); const targetCodes = payload.map(item => item.Code.toString().trim()); const orderMap = {}; payload.forEach(item => orderMap[item.Code.toString().trim()] = item.OrderNo); // 1. 處理總表 (GiftStatus) const statusSheet = SS.getSheetByName("GiftStatus"); const statusData = statusSheet.getDataRange().getValues(); const statusIndexMap = {}; // 代碼 -> 行號(0-indexed) for (let i = 1; i < statusData.length; i++) { statusIndexMap[statusData[i][3].toString().trim()] = i; } let count = 0; let sheetsToUpdate = {}; // 供應商_項目 -> [代碼清單] targetCodes.forEach(code => { const idx = statusIndexMap[code]; if (idx !== undefined) { const row = statusData[idx]; const orderNo = orderMap[code]; const subSheetName = row[1] + "_" + row[2]; if (row[6] && row[6] !== "") { // 二次登記 row[9] = orderNo; row[10] = user; row[11] = now; } else { // 首次登記 row[5] = "已使用"; row[6] = orderNo; row[7] = user; row[8] = now; } if (!sheetsToUpdate[subSheetName]) sheetsToUpdate[subSheetName] = []; sheetsToUpdate[subSheetName].push(code); count++; } }); if (count > 0) { // 批次寫回總表受影響區域 (如果代碼分散，全量寫回仍是最快，但我們優化了索引尋找) statusSheet.getDataRange().setValues(statusData); // 2. 處理分表 (精準更新) for (const sheetName in sheetsToUpdate) { const subSheet = SS.getSheetByName(sheetName); if (subSheet) { const subData = subSheet.getDataRange().getValues(); const subCodes = sheetsToUpdate[sheetName]; const subIndexMap = {}; for (let j = 1; j < subData.length; j++) { subIndexMap[subData[j][3].toString().trim()] = j; } subCodes.forEach(code => { const sIdx = subIndexMap[code]; if (sIdx !== undefined) { const sRow = subData[sIdx]; const orderNo = orderMap[code]; if (sRow[6] && sRow[6] !== "") { sRow[9] = orderNo; sRow[10] = user; sRow[11] = now; } else { sRow[5] = "已使用"; sRow[6] = orderNo; sRow[7] = user; sRow[8] = now; } } }); subSheet.getDataRange().setValues(subData); } } // 3. 寫入流水帳 const recordSheet = SS.getSheetByName("Records"); const r = data.recordInfo; if (r) { recordSheet.appendRow([now, r.supplier, r.gift, r.start, r.end, r.count, user, r.orderNo, r.expiry, "正常", "使用"]); } } return jsonResponse({ result: "success", count: count }); } if (mode === 'activate' || mode === 'activateRange') { const sheet = SS.getSheetByName("GiftStatus"); const dataRange = sheet.getDataRange(); const statusData = dataRange.getValues(); const codes = mode === 'activate' ? new Set(data.payload.map(c => c.toString().trim())) : null; const start = data.start, end = data.end; let count = 0; let sheetsToUpdate = new Set(); for (let i = 1; i < statusData.length; i++) { const currentCode = statusData[i][3].toString().trim(); const shouldActivate = mode === 'activate' ? codes.has(currentCode) : (currentCode >= start && currentCode <= end); if (shouldActivate) { statusData[i][4] = "已啟動"; sheetsToUpdate.add(statusData[i][1] + "_" + statusData[i][2]); count++; } } if (count > 0) { dataRange.setValues(statusData); sheetsToUpdate.forEach(sheetName => { const subSheet = SS.getSheetByName(sheetName); if (subSheet) { const subRange = subSheet.getDataRange(); const subData = subRange.getValues(); for (let j = 1; j < subData.length; j++) { const subCode = subData[j][3].toString().trim(); const shouldSubAct = mode === 'activate' ? codes.has(subCode) : (subCode >= start && subCode <= end); if (shouldSubAct) subData[j][4] = "已啟動"; } subRange.setValues(subData); } }); } return jsonResponse({ result: "success", count: count }); } if (mode === 'markDeleted') { const recordSheet = SS.getSheetByName("Records"); const statusSheet = SS.getSheetByName("GiftStatus"); const statusRange = statusSheet.getDataRange(); const statusData = statusRange.getValues(); const rowIdx = data.index + 2; const recordData = recordSheet.getRange(rowIdx, 1, 1, 11).getValues()[0]; const startCode = recordData[3]; const endCode = recordData[4]; const type = recordData[10]; recordSheet.getRange(rowIdx, 10).setValue("已登記刪除"); if (type === "使用") { const codesToReset = new Set(getRangeCodes(startCode, endCode).map(c => c.toString().trim())); let sheetsToUpdate = new Set(); let count = 0; for (let i = 1; i < statusData.length; i++) { const code = statusData[i][3].toString().trim(); if (codesToReset.has(code)) { statusData[i][5] = "未使用"; statusData[i][6] = ""; statusData[i][7] = ""; statusData[i][8] = ""; sheetsToUpdate.add(statusData[i][1] + "_" + statusData[i][2]); count++; } } if (count > 0) { statusRange.setValues(statusData); sheetsToUpdate.forEach(sheetName => { const subSheet = SS.getSheetByName(sheetName); if (subSheet) { const subRange = subSheet.getDataRange(); const subData = subRange.getValues(); for (let j = 1; j < subData.length; j++) { if (codesToReset.has(subData[j][3].toString().trim())) { subData[j][5] = "未使用"; subData[j][6] = ""; subData[j][7] = ""; subData[j][8] = ""; } } subRange.setValues(subData); } }); } } return jsonResponse({ result: "success" }); } return jsonResponse({ result: "error", message: "Unknown POST action" }); } catch (err) { return jsonResponse({ result: "error", message: err.toString() }); } } function getOrCreateSubSheet(name) { let sheet = SS.getSheetByName(name); if (!sheet) { sheet = SS.insertSheet(name); sheet.appendRow(["批次", "供應商", "禮券", "代碼", "啟動狀態", "使用狀態", "訂單號", "使用者", "時間", "訂單號2", "使用者2", "時間2"]); sheet.getRange("1:1").setFontWeight("bold").setBackground("#f3f3f3"); sheet.setFrozenRows(1); } return sheet; } function getRangeCodes(start, end) { var sStr = start.toString().trim(); var eStr = end.toString().trim(); if(!sStr || !eStr) return []; var sMatch = sStr.match(/^([A-Za-z]*)(.*)$/); var eMatch = eStr.match(/^([A-Za-z]*)(.*)$/); var prefix = sMatch[1] || ""; var sNumStr = sMatch[2] || "0"; var eNumStr = eMatch[2] || "0"; var sVal = parseInt(sNumStr, 10); var eVal = parseInt(eNumStr, 10); var padLen = sNumStr.length; var codes = []; for (var i = sVal; i <= eVal; i++) { codes.push(prefix + i.toString().padStart(padLen, '0')); } return codes; } function jsonResponse(obj) { return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON);
